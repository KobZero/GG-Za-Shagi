#include<iomanip>
using namespace std;
int main()
{
	SetConsoleOutputCP(1251);
	SetConsoleCP(1251);
	ofstream fout;
	ifstream fin;
	fout.open("C:\\Users\\pupil\\Desktop\\ttt\\file.txt");
	fin.open("C:\\Users\\pupil\\Desktop\\ttt\\input.txt");
	int ch;
	string str;
	vector<string> names;
	vector<int> ages;
	while (!fin.eof())
	{
		fin >> str;
		fin >> ch;
		names.push_back(str);
		ages.push_back(ch);
	}
	for (int i=0; i < names.size()-1; i++)
	{
		for (int j = 0; j < names.size() - 1; j++)
		{
			swap(ages[j], ages[j + 1]);
			swap(names[j], names[j + 1]);
		}
	}
	fout.close();
	fin.close();
	system("pause");
}
